*Cerate DTO to create and upate the AdminArea* 

1-class : AdminAreaCreateRequest

@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
public class AdminAreaCreateRequest {
	
	@NotBlank
	@Pattern(regexp = "^[0-9]{2}(?:-[0-9]{2}){0,3}$", message = "code must look like 12 or 12-03-07-02")
	private String code;
	
	@NotNull
	private AdminLevel adminLevel;
	
	private String parentCode;
	
	@NotBlank
	private String nameKh;
	
	@NotBlank
	private String nameEn;

}

2-class : AdminAreaUpdateRequest


@Data
@Builder

@AllArgsConstructor
public class AdminAreaUpdateRequest {
	
	@NotBlank
	private String nameKh;
	@NotBlank
	private String nameEn;

}
